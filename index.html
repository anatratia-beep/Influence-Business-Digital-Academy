// ═══════════════════════════════════════════════════════════ // API — Parse SDK (gère CORS automatiquement) // ═══════════════════════════════════════════════════════════ function initParse() { Parse.initialize(CONFIG.APP_ID, CONFIG.JAVASCRIPT_KEY); Parse.serverURL = CONFIG.API_URL; } // ── Auth ──────────────────────────────────────────────────── async function login(username, password) { try { initParse(); const user = await Parse.User.logIn(username, password); return { objectId: user.id, username: user.get("username"), fullName: user.get("fullName"), classe: user.get("classe"), role: user.get("role"), instagram: user.get("instagram") || "", tiktok: user.get("tiktok") || "", facebook: user.get("facebook") || "", youtube: user.get("youtube") || "", sessionToken: user.getSessionToken(), }; } catch(e) { console.error("Login error:", e); return null; } } async function register(username, password, fullName, salle, horaire) { try { initParse(); const user = new Parse.User(); user.set("username", username); user.set("password", password); user.set("fullName", fullName); user.set("classe", `${salle} — ${horaire}`); user.set("salle", salle); user.set("horaire", horaire); user.set("role", "etudiant"); await user.signUp(); return true; } catch(e) { console.error("Register error:", e); return null; } } async function changePassword(objectId, newPassword, sessionToken) { try { initParse(); const current = await Parse.User.become(sessionToken); current.set("password", newPassword); await current.save(); return true; } catch(e) { return false; } } // ── Tâches ────────────────────────────────────────────────── async function getAllTasks() { try { initParse(); const Task = Parse.Object.extend("Task"); const query = new Parse.Query(Task); query.descending("createdAt"); query.limit(200); const results = await query.find(); const tasks = results.map(t => ({ objectId: t.id, title: t.get("title"), description: t.get("description") || "", autoGenerated: t.get("autoGenerated"), })); const prof = tasks.filter(t => t.autoGenerated !== true); const student = tasks.filter(t => t.autoGenerated === true); return [...prof, ...student]; } catch(e) { console.error(e); return []; } } async function createTask(title, description) { try { initParse(); const Task = Parse.Object.extend("Task"); const t = new Task(); t.set("title", title); t.set("description", description); t.set("autoGenerated", false); await t.save(); return true; } catch(e) { return false; } } async function createTaskForPublication(studentName, networkName, link) { try { initParse(); const Task = Parse.Object.extend("Task"); const t = new Task(); t.set("title", `Liker/commenter le ${networkName} de ${studentName}`); t.set("description", link); t.set("autoGenerated", true); t.set("studentName", studentName); t.set("network", networkName); t.set("publicationLink", link); await t.save(); return true; } catch(e) { return false; } } async function deleteTask(taskId) { try { initParse(); const Task = Parse.Object.extend("Task"); const query = new Parse.Query(Task); const t = await query.get(taskId); await t.destroy(); return true; } catch(e) { return false; } } // ── Soumissions ────────────────────────────────────────────── async function getAllSubmissions() { try { initParse(); const Sub = Parse.Object.extend("Submission"); const query = new Parse.Query(Sub); query.include("task"); query.include("student"); query.limit(500); const results = await query.find(); return results.map(s => ({ objectId: s.id, proofLink: s.get("proofLink") || "", commentUsed: s.get("commentUsed") || "", student: { objectId: s.get("student")?.id, fullName: s.get("student")?.get("fullName"), username: s.get("student")?.get("username"), }, task: { objectId: s.get("task")?.id, title: s.get("task")?.get("title"), }, })); } catch(e) { console.error(e); return []; } } async function getSubmissionsByStudent(studentId) { try { initParse(); const Sub = Parse.Object.extend("Submission"); const User = Parse.Object.extend("_User"); const userPtr = User.createWithoutData(studentId); const query = new Parse.Query(Sub); query.equalTo("student", userPtr); query.include("task"); query.limit(200); const results = await query.find(); return results.map(s => ({ objectId: s.id, task: { objectId: s.get("task")?.id, title: s.get("task")?.get("title") }, })); } catch(e) { console.error(e); return []; } } async function submitTask(studentId, taskId, proofLink, commentUsed, sessionToken) { try { initParse(); const current = await Parse.User.become(sessionToken); const Sub = Parse.Object.extend("Submission"); const s = new Sub(); const User = Parse.Object.extend("_User"); const Task = Parse.Object.extend("Task"); s.set("student", User.createWithoutData(studentId)); s.set("task", Task.createWithoutData(taskId)); s.set("status", "terminé"); s.set("proofLink", proofLink); s.set("commentUsed", commentUsed); await s.save(null, { sessionToken: current.getSessionToken() }); return true; } catch(e) { console.error("Submit error:", e); return false; } } // ── Étudiants ──────────────────────────────────────────────── async function getAllStudents() { try { initParse(); const query = new Parse.Query(Parse.User); query.equalTo("role", "etudiant"); query.limit(200); const results = await query.find(); return results.map(s => ({ objectId: s.id, username: s.get("username"), fullName: s.get("fullName") || "", classe: s.get("classe") || "", instagram: s.get("instagram") || "", tiktok: s.get("tiktok") || "", facebook: s.get("facebook") || "", youtube: s.get("youtube") || "", })); } catch(e) { console.error(e); return []; } } async function updateUser(objectId, data, sessionToken) { try { initParse(); const current = await Parse.User.become(sessionToken); Object.entries(data).forEach(([k, v]) => current.set(k, v)); await current.save(); return true; } catch(e) { return false; } } // ── Commentaires ───────────────────────────────────────────── async function getPresetComments() { try { initParse(); const PC = Parse.Object.extend("PresetComment"); const query = new Parse.Query(PC); const res = await query.find(); return res.map(c => ({ text: c.get("text") || "" })); } catch(e) { return []; } }
